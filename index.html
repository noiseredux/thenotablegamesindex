// CONFIGURATION const GOOGLE_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycby3_IE8KU_qeQ4JCgTnoaF1w--vPUemtxHjW5Aq3aF_cUN8eqmYny7xlAlbT-wMp4Qj/exec'; const ALLOWED_PLATFORMS = [130, 37, 24, 29, 32, 35, 4, 21]; function displayGames(gamesList) { const content = document.getElementById('content'); let filtered = gamesList.filter(game => { // 1. Platform validation const hasPlatform = game.platforms && game.platforms.some(p => ALLOWED_PLATFORMS.includes(p)); if (!hasPlatform) return false; // 2. Category Filter (Allows Main games [0], Expansions [4], Remakes [8], and Ports [11]) const categoryNum = (game.category !== undefined && game.category !== null) ? Number(game.category) : 0; if (categoryNum !== 0 && categoryNum !== 4 && categoryNum !== 8 && categoryNum !== 11) return false; // 3. Basic containment filter for standard entries if (categoryNum === 0 || categoryNum === 4) { if (game.parent_game !== undefined && game.parent_game !== null) return false; if (game.version_parent !== undefined && game.version_parent !== null) return false; } // 4. Faux Content Filter if (game.keywords && game.keywords.length > 0) { const isFauxContent = game.keywords.some(k => k.slug === 'fangame' || k.slug === 'unofficial'); if (isFauxContent) return false; } return true; }); if (!filtered.length) { content.innerHTML = '
No records found for this year.
'; return; } // Build UI Grid items loop content.innerHTML = filtered.map(game => { const scoreDisplay = game.total_rating ? `${Math.round(game.total_rating)}%` : 'N/A'; const coverUrl = game.cover ? `https://images.igdb.com/igdb/image/upload/t_cover_big/${game.cover}.jpg` : 'https://via.placeholder.com/264x352?text=No+Cover'; return `
${game.name}
Score: ${scoreDisplay}
`; }).join(''); }
